49

151 int BareHamiltonian::CalcSign(int i,int j,myint a) const
152 {
153  int sign;
154 
155  // count the number of set bits between i and j in ket a
156  sign = CountBits(( ((1<<j) - 1) ^ ((1<<(i+1)) - 1) ) & a);
157 
158  // when uneven, we get a minus sign
159  if( sign & 0x1 )
160  return -1;
161  else
162  return +1;
163 }

287 double BareHamiltonian::LanczosDiagonalize(int m)
288 {
289  if(!m)
290  m = 10;
291 
292  std::vector<double> a(m,0);
293  std::vector<double> b(m,0);
294 
295  double *qa = new double [dim];
296  double *qb = new double [dim];
297 
298  double E = 1;
299 
300  int i;
301 
302  srand(time(0));
303 
304  for(i=0;i<dim;i++)
305  {
306  qa[i] = 0;
307  qb[i] = rand()*1.0/RAND_MAX;
308  }
309 
310  int incx = 1;
311 
312  double norm = 1.0/sqrt(ddot_(&dim,qb,&incx,qb,&incx));
313 
314  dscal_(&dim,&norm,qb,&incx);
315 
316  norm = 1;
317 
318  double *f1 = qa;
319  double *f2 = qb;
320  double *tmp;
321 
322  double alpha;
323 
324  std::vector<double> acopy(a);
325  std::vector<double> bcopy(b);
326 
327  i = 1;
328 
329  while(fabs(E-acopy[0]) > 1e-4)
330  {
331  E = acopy[0];
332 
333  for(;i<m;i++)
334  {
335  alpha = -b[i-1];
336  dscal_(&dim,&alpha,f1,&incx);
337 
338  mvprod(f2,f1,norm);
339 
340  a[i-1] = ddot_(&dim,f1,&incx,f2,&incx);
341 
342  alpha = -a[i-1];
343  daxpy_(&dim,&alpha,f2,&incx,f1,&incx);
344 
345  b[i] = sqrt(ddot_(&dim,f1,&incx,f1,&incx));
346 
347  if( fabs(b[i]) < 1e-10 )
348  break;
349 
350  alpha = 1.0/b[i];
351 
352  dscal_(&dim,&alpha,f1,&incx);
353 
354  tmp = f2;
355  f2 = f1;
356  f1 = tmp;
357  }
358 
359  acopy = a;
360  bcopy = b;
361 
362  char jobz = 'N';
363  int info;
364 
365  dstev_(&jobz,&m,acopy.data(),&bcopy.data()[1],&alpha,&m,&alpha,&info);
366 
367  if(info != 0)
368  std::cerr << "Error in Lanczos" << std::endl;
369 
370  m += 10;
371  a.resize(m);
372  b.resize(m);
373  }
374 
375  delete [] qa;
376  delete [] qb;
377 
378  return acopy[0];
379 }

410 double BareHamiltonian::arpackDiagonalize()
411 {
412  // dimension of the matrix
413  int n = dim;
414 
415  // number of eigenvalues to calculate
416  int nev = 2;
417 
418  // reverse communication parameter, must be zero on first iteration
419  int ido = 0;
420  // standard eigenvalue problem A*x=lambda*x
421  char bmat = 'I';
422  // calculate the smallest algebraic eigenvalue
423  char which[] = {'S','A'};
424  // calculate until machine precision
425  double tol = 0;
426 
427  // the residual vector
428  double *resid = new double[n];
429 
430  // the number of columns in v: the number of lanczos vector
431  // generated at each iteration, ncv <= n
432  // We use the answer to life, the universe and everything, if possible
433  int ncv = 16;
434 
435  if( n < ncv )
436  ncv = n;
437 
438  // v containts the lanczos basis vectors
439  int ldv = n;
440  double *v = new double[ldv*ncv];
441 
442  int *iparam = new int[11];
443  iparam[0] = 1; // Specifies the shift strategy (1->exact)
444  iparam[2] = 3*n; // Maximum number of iterations
445  iparam[6] = 1; /* Sets the mode of dsaupd.
446  1 is exact shifting,
447  2 is user-supplied shifts,
448  3 is shift-invert mode,
449  4 is buckling mode,
450  5 is Cayley mode. */
451 
452  int *ipntr = new int[11]; /* Indicates the locations in the work array workd
453  where the input and output vectors in the
454  callback routine are located. */
455 
456  // array used for reverse communication
457  double *workd = new double[3*n];
458 
459  int lworkl = ncv*(ncv+8); /* Length of the workl array */
460  double *workl = new double[lworkl];
461 
462  // info = 0: random start vector is used
463  int info = 0; /* Passes convergence information out of the iteration
464  routine. */
465 
466  // rvec == 0 : calculate only eigenvalue
467  // rvec > 0 : calculate eigenvalue and eigenvector
468  int rvec = 0;
469 
470  // how many eigenvectors to calculate: 'A' => nev eigenvectors
471  char howmny = 'A';
472 
473  int *select;
474  // when howmny == 'A', this is used as workspace to reorder the eigenvectors
475  if( howmny == 'A' )
476  select = new int[ncv];
477 
478  // This vector will return the eigenvalues from the second routine, dseupd.
479  double *d = new double[nev];
480 
481  double *z = 0;
482 
483  if( rvec )
484  z = new double[n*nev];
485 
486  // not used if iparam[6] == 1
487  double sigma;
488 
489  // first iteration
490  dsaupd_(&ido, &bmat, &n, &which[0], &nev, &tol, resid, &ncv, v, &ldv, iparam, ipntr, workd, workl, &lworkl, &info);
491 
492  while( ido != 99 )
493  {
494  // matrix-vector multiplication
495  mvprod(workd+ipntr[0]-1, workd+ipntr[1]-1,0);
496 
497  dsaupd_(&ido, &bmat, &n, &which[0], &nev, &tol, resid, &ncv, v, &ldv, iparam, ipntr, workd, workl, &lworkl, &info);
498  }
499 
500  if( info < 0 )
501  std::cerr << "Error with dsaupd, info = " << info << std::endl;
502  else if ( info == 1 )
503  std::cerr << "Maximum number of Lanczos iterations reached." << std::endl;
504  else if ( info == 3 )
505  std::cerr << "No shifts could be applied during implicit Arnoldi update, try increasing NCV." << std::endl;
506 
507  dseupd_(&rvec, &howmny, select, d, z, &ldv, &sigma, &bmat, &n, which, &nev, &tol, resid, &ncv, v, &ldv, iparam, ipntr, workd, workl, &lworkl, &info);
508 
509  if ( info != 0 )
510  std::cerr << "Error with dseupd, info = " << info << std::endl;
511 
512  // use something to store the result before deleting...
513  sigma = d[0];
514  std::cout << "0: " << d[0] << std::endl;
515  std::cout << "1: " << d[1] << std::endl;
516 
517  delete [] resid;
518  delete [] v;
519  delete [] iparam;
520  delete [] ipntr;
521  delete [] workd;
522  delete [] workl;
523  delete [] d;
524 
525  if( rvec )
526  delete [] z;
527 
528  if( howmny == 'A' )
529  delete [] select;
530 
531  return sigma;
532 }

630 void BareHamiltonian::Diagonalize(int dim, double *mat, double *eigs, bool calc_eigenvectors)
631 {
632  if(!mat || !eigs)
633  throw;
634 
635  char jobz;
636 
637  if(calc_eigenvectors)
638  jobz = 'V';
639  else
640  jobz = 'N';
641 
642  char uplo = 'U';
643 
644  int lwork, liwork;
645 
646  if(calc_eigenvectors)
647  {
648  lwork = 6*dim+1+2*dim*dim;
649  liwork = 3+5*dim;
650  } else
651  {
652  lwork = 2*dim+1;
653  liwork = 1;
654  }
655 
656  double *work = new double[lwork];
657 
658  int *iwork = new int[liwork];
659 
660  int info;
661 
662  dsyevd_(&jobz, &uplo, &dim, mat, &dim, eigs, work, &lwork, iwork, &liwork, &info);
663 
664  if(info != 0)
665  std::cerr << "Calculating eigenvalues failed..." << std::endl;
666 
667  delete [] work;
668  delete [] iwork;
669 }
