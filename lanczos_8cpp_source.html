48

51 
52  srand(time(0));
53

55  for(int j=i;j<size;j++)
56  matrix(i,j) = matrix(j,i) = rand()*1.0/RAND_MAX;
57 
58  vector<double> a(m,0);
59  vector<double> b(m,0);
60 
61  int i;
62 
63  vec qa(size);
64  vec qb(size);
65 
66  qa.zeros();
67 
68  qb = randu(size);
69  double norm = sqrt(dot(qb,qb));
70  qb *= 1.0/norm;
71 
72  vec *f1 = &qa;
73  vec *f2 = &qb;
74  vec *tmp;
75 
76  for(i=1;i<m;i++)
77  {
78  (*f1) *= -b[i-1];
79  (*f1) += matrix*(*f2);
80 
81  a[i-1] = dot(*f1,*f2);
82 
83  (*f1) -= a[i-1]*(*f2);
84 
85  b[i] = sqrt(dot(*f1,*f1));
86 
87  if( fabs(b[i]) < 1e-10 )
88  break;
89 
90  (*f1) /= b[i];
91  tmp = f2;
92  f2 = f1;
93  f1 = tmp;
94  }
95

97 
98  T.zeros();

100 
101  for(int j=1;j<i;j++)
102  {
103  T(j,j) = a[j];
104  T(j,j-1) = T(j-1,j) = b[j];
105  }
106 
107  vec eigs1 = eig_sym(T);
108  vec eigs2 = eig_sym(matrix);
109 
110  cout << eigs1[0] << endl;

115 }
